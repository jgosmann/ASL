    \begin{block}{Blocktitel}
        Blocktext
  \end{block}

  \begin{exampleblock}{Beispielblocktitel}
        Beispielblocktext
  \end{exampleblock}

  \begin{alertblock}{Warnungsblocktitel}
        Warnungsblocktext
  \end{alertblock}
\end{frame}
